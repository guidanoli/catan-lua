Module util.class

Defines classes and how to instantiate them.

local Class = require "util.class"
local Dog = Class "Dog"
local bob = Dog:__new{name = "Bob"}
print(bob)      --> Dog: 0x55b03f40fa70
print(bob.name) --> Bob
